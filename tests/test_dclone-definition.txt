defmod test_dclone UnitTest autocreate (
 {
	
 	subtest 'dclone / restore verify ' => sub {
		plan tests => 1;
		$defs{global}{logMethod} = \&main::Log3;	

		my %copyOfTargetHash = %{ dclone( $defs{$target} ) };
		$defs{global}{logMethod} = \&main::Log;	

		is(exists($copyOfTargetHash->{logMethod}, 1, "cloned hash is defined ");
		$defs{$target} = \%copyOfTargetHash;
		
	}; 

 };
);
